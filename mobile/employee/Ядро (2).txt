  responses:

        status: { $ref: '#/components/schemas/OperationStatus' }
      required: [status]

    OperationStatus:
      type: string
      enum: [CREATED, IN_PROCESS, SUCCESS, REJECTED]

    AccountActionType:
      type: string
      enum: [OPEN_ACCOUNT, CLOSE_ACCOUNT, TRANSFER_RECEIVED, TRANSFER_SENT]

      type: object
      properties:
